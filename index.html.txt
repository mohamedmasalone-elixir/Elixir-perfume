    <script>
        // ----------------------
        // إدارة السلة
        // ----------------------
        let cart = [];

        function addToCart(id, name, price) {
            const existing = cart.find(item => item.id === id);
            if (existing) {
                existing.qty += 1;
            } else {
                cart.push({ id, name, price, qty: 1 });
            }
            renderCart();
        }

        function removeFromCart(id) {
            cart = cart.filter(item => item.id !== id);
            renderCart();
        }

        function renderCart() {
            const tbody = document.getElementById('cart-items');
            const countSpan = document.getElementById('cart-count');
            const totalSpan = document.getElementById('cart-total');

            tbody.innerHTML = '';

            if (cart.length === 0) {
                tbody.innerHTML = '<tr><td colspan="5">السلة فارغة حالياً</td></tr>';
                countSpan.textContent = '0';
                totalSpan.textContent = '0';
                return;
            }

            let total = 0;
            let count = 0;

            cart.forEach(item => {
                const row = document.createElement('tr');
                const itemTotal = item.price * item.qty;
                total += itemTotal;
                count += item.qty;

                row.innerHTML = `
                    <td>${item.name}</td>
                    <td>${item.qty}</td>
                    <td>${item.price} جنيه</td>
                    <td>${itemTotal} جنيه</td>
                    <td><button class="btn secondary" onclick="removeFromCart('${item.id}')">حذف</button></td>
                `;

                tbody.appendChild(row);
            });

            countSpan.textContent = count;
            totalSpan.textContent = total;
        }

        // ----------------------
        // وسيلة الدفع / إرسال الطلب
        // ----------------------

        // ضع هنا رابط صفحة الدفع الخاصة بك (PayPal, صفحة بنك، فوري، الخ)
        // مثلاً: const PAYMENT_URL_BASE = "https://your-payment-link.com/pay";
        const PAYMENT_URL_BASE = "https://example.com/pay"; // عدّل هذا الرابط

        function checkout() {
            if (cart.length === 0) {
                alert('السلة فارغة. أضف بعض المنتجات أولاً.');
                return;
            }

            const total = cart.reduce((sum, item) => sum + item.price * item.qty, 0);
            const details = cart.map(item => `${item.name} x${item.qty}`).join(', ');

            // مثال 1: فتح صفحة دفع عامة مع المبلغ
            // const url = `${PAYMENT_URL_BASE}?amount=${encodeURIComponent(total)}&details=${encodeURIComponent(details)}`;
            // window.open(url, '_blank');

            // مثال 2: إرسال الطلب عبر واتساب (أسهل شيء بدون برمجة سيرفر)
            const phone = '201000000000'; // ضع رقم الواتساب الخاص بك بدون +
            const msg = `طلب جديد من المتجر:%0A${details}%0Aالإجمالي: ${total} جنيه`;
            const waUrl = `https://wa.me/${phone}?text=${msg}`;
            window.open(waUrl, '_blank');
        }

        // ----------------------
        // إرسال نموذج التواصل عبر واتساب
        // ----------------------
        function sendContact(e) {
            e.preventDefault();
            const name = document.getElementById('name').value.trim();
            const phone = document.getElementById('phone').value.trim();
            const message = document.getElementById('message').value.trim();

            const waPhone = '201000000000'; // نفس رقم الواتساب أو رقم مختلف
            const text = `استفسار جديد من الموقع:%0Aالاسم: ${name}%0Aالهاتف: ${phone}%0Aالرسالة: ${message}`;
            const url = `https://wa.me/${waPhone}?text=${text}`;
            window.open(url, '_blank');
        }
    </script>
